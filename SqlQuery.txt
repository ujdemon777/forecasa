user schema: 

CREATE TABLE IF NOT EXISTS user(
	id INT AUTO_INCREMENT PRIMARY KEY,
	name VARCHAR(128),
	email VARCHAR(128) UNIQUE NOT NULL,
	password VARCHAR(128) NOT NULL,
	phone VARCHAR(128),
	role VARCHAR(128) DEFAULT 'user' NOT NULL,
	status VARCHAR(128) NOT NULL,
	is_authenticated BOOLEAN DEFAULT FALSE,
	created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
) ENGINE = INNODB;



CREATE TABLE IF NOT EXISTS contact (
    id INT AUTO_INCREMENT PRIMARY KEY,
    first_name VARCHAR(255),
    last_name VARCHAR(255),
    email VARCHAR(255),
    primary_contact VARCHAR(255),
    secondary_contact VARCHAR(255),
    linkedIn VARCHAR(255),
    contact_type ENUM('primary', 'secondary'), 
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    company_id INT,
    CONSTRAINT contact_ibfk_1
    FOREIGN KEY (company_id) REFERENCES company(id)
)ENGINE = INNODB;


CREATE TABLE IF NOT EXISTS blobs (
    id INT AUTO_INCREMENT PRIMARY KEY,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    file_name VARCHAR(255) UNIQUE NOT NULL,
    source VARCHAR(255) NOT NULL,
    meta_data JSON NOT NULL,
    status VARCHAR(64) NOT NULL,
    project_label VARCHAR(255) NOT NULL,
    user_id INT NOT NULL,
    CONSTRAINT blobs_ibfk_1
    FOREIGN KEY (user_id) REFERENCES user(id)
) ENGINE = INNODB;



CREATE TABLE IF NOT EXISTS business_credential (
    id INT AUTO_INCREMENT PRIMARY KEY,
    business_id INT NOT NULL,
    source VARCHAR(128) NOT NULL,
    api_key VARBINARY(1000),
    fernet_key VARBINARY(1000),
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
) ENGINE = INNODB;
